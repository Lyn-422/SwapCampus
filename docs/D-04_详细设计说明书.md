```python
    """统一异常处理器
    
    将 DRF 异常和 Django Http404 包装为统一格式
    """
    response = drf_exception_handler(exc, context)
    if response is not None:
        # 根据状态码映射错误码

            400: "BAD_REQUEST",
            401: "UNAUTHORIZED",
            403: "FORBIDDEN",
            404: "NOT_FOUND",
```

```python
    """自定义用户模型"""
    # 覆盖默认主键为 UUID
    id = models.UUIDField(primary_key=True, default=uuid.uuid4)
    
    # 扩展字段
    nickname = models.CharField(max_length=50, blank=True)  # 昵称
    avatar = models.ImageField(upload_to="avatars/%Y/%m/", blank=True)
    credit_score = models.IntegerField(default=100, db_index=True)  # 信用分
    campus = models.CharField(max_length=30, blank=True)  # 校区
    bio = models.TextField(max_length=500, blank=True)  # 个人简介
    
    @property
    def credit_level(self) -> str:
        """信用等级计算"""
        if self.credit_score >= 150: return "excellent"
        elif self.credit_score >= 100: return "good"
        elif self.credit_score >= 60: return "fair"
        else: return "poor"
    
    @property
    def is_trusted_seller(self) -> bool:
        """可信卖家判断"""
        if not self.is_active or self.credit_score < 150:
            return False
        # 180天内无违规
        recent_violation = CreditRecord.objects.filter(
            user=self, reason=CreditRecord.ChangeReason.VIOLATION,
            created_at__gte=timezone.now() - timedelta(days=180)
        ).exists()
        if recent_violation: return False
        # 完成交易≥3单
        completed = Order.objects.filter(
            seller=self, status=Order.Status.COMPLETED
        ).count()
        return completed >= 3
```

```python
@transaction.atomic
def add_credit_record(user: User, change: int, reason: str, 
                      description: str = "", related_order=None) -> CreditRecord:
    """原子化地添加积分变更记录
    
    使用 select_for_update 加行锁，防止并发积分覆盖问题
    """
    # 锁定用户行
    locked_user = User.objects.select_for_update().get(pk=user.pk)
    new_score = max(0, locked_user.credit_score + change)  # 积分下限为0
    locked_user.credit_score = new_score
    locked_user.save(update_fields=["credit_score"])
    
    # 创建积分记录
    record = CreditRecord.objects.create(
        user=user, change=change, reason=reason,
        description=description, score_after=new_score,
        related_order=related_order
    )
    
    # 创建积分变动通知
    create_notification(user, "credit_change", "积分变动",
        f"你的信用分 {sign}{change}，当前 {new_score} 分",
        related_order=related_order)
    
    return record
```

#### 3.3.3 核心业务逻辑

##### 状态转换函数

```python
@db_transaction.atomic
def transition_order(order: Order, new_status_str: str, user, **kwargs) -> Order:
    """订单状态转换（核心方法）
    
    Args:
        order: 订单实例
        new_status_str: 目标状态
        user: 执行操作的用户
        **kwargs: 额外字段（cancel_reason, meet_time等）
    
    Raises:
        ValueError: 非法状态转换
    """
    current_status = order.status
    allowed = ORDER_TRANSITIONS.get(current_status, [])
    
    if new_status_str not in allowed:
        raise ValueError(f"不允许从 {current_status} 转换到 {new_status_str}")
    
    # 锁定订单行（并发安全）
    order = Order.objects.select_for_update().get(pk=order.pk)
    order.status = new_status_str
    
    # 更新额外字段
    for field, value in kwargs.items():
        if hasattr(order, field):
            setattr(order, field, value)
    
    # 触发副作用（通知、积分、商品状态变更）
    if new_status_str == Order.Status.ACCEPTED:
        create_notification(order.buyer, "order_update", 
            "订单已接受", "卖家已接受你的订单，请准备面交")
    
    elif new_status_str == Order.Status.COMPLETED:
        order.completed_at = timezone.now()
        # 双方各+5信用分
        _add_credit(order.buyer, "order_complete", f"完成交易", order)
        _add_credit(order.seller, "order_complete", f"完成交易", order)
        # 商品标记已售出
        Product.objects.filter(pk=order.product_id).update(status=Product.Status.SOLD)
    
    order.save()
    return order
```

##### 面交确认机制

```python
def generate_face_confirm_code() -> str:
    """生成6位随机数字确认码"""
    return str(random.randint(100000, 999999))

@db_transaction.atomic
def create_face_confirm(order: Order, created_by) -> FaceConfirm:
    """生成面交确认码（卖家操作）"""
    if created_by.id != order.seller_id:
        raise ValueError("只有卖家可以生成面交确认码")
    
    # 复用未过期的确认码
    existing = FaceConfirm.objects.filter(
        order=order, status=FaceConfirm.Status.PENDING
    ).first()
    if existing:
        return existing
    
    code = generate_face_confirm_code()
    face_confirm = FaceConfirm.objects.create(
        order=order, confirm_code=code, created_by=created_by
    )
    
    # 订单进入面交确认状态
    order.status = Order.Status.FACE_CONFIRM
    order.save(update_fields=["status"])
    
    return face_confirm

@db_transaction.atomic
def verify_face_confirm(order: Order, code: str, confirmed_by) -> Order:
    """验证面交确认码（买家操作）"""
    if confirmed_by.id != order.buyer_id:
        raise ValueError("只有买家可以验证确认码")
    
    face_confirm = FaceConfirm.objects.select_for_update().get(
        order=order, status=FaceConfirm.Status.PENDING
    )
    
    if face_confirm.confirm_code != code:
        raise ValueError("确认码错误")
    
    face_confirm.status = FaceConfirm.Status.CONFIRMED
    face_confirm.confirmed_by = confirmed_by
    face_confirm.confirmed_at = timezone.now()
    face_confirm.save()
    
    # 订单完成
    return transition_order(order, Order.Status.COMPLETED, confirmed_by)
```

##### 评价系统

```python
@db_transaction.atomic
def create_review(order: Order, reviewer, rating: int, content: str = "") -> Review:
    """创建交易评价
    
    规则：
    1. 只能评价已完成的订单
    2. 30天评价有效期
    3. 最多2次评价（首次+追评）
    4. 仅首次评价触发信用分变动
    """
    if order.status != Order.Status.COMPLETED:
        raise ValueError("只能评价已完成的订单")
    
    # 检查30天时限
    if timezone.now() > order.completed_at + timedelta(days=30):
        raise ValueError("订单已完成超过30天，无法评价")
    
    # 确定评价类型
    if reviewer.id == order.buyer_id:
        review_type = Review.ReviewType.BUYER_TO_SELLER
        reviewee = order.seller
        count_field = "buyer_review_count"
    else:
        review_type = Review.ReviewType.SELLER_TO_BUYER
        reviewee = order.buyer
        count_field = "seller_review_count"
    
    # 检查评价次数
    current_count = getattr(order, count_field)
    if current_count >= 2:
        raise ValueError("评价次数已达上限")
    
    is_first_review = current_count == 0
    
    review = Review.objects.create(
        order=order, reviewer=reviewer, reviewee=reviewee,
        rating=rating, content=content, review_type=review_type
    )
    
    # 递增评价次数
    setattr(order, count_field, current_count + 1)
    order.save(update_fields=[count_field])
    
    # 仅首次评价触发信用分变动
    if is_first_review:
        if rating >= 4:
            _add_credit(reviewee, "good_review", f"获得好评", order)
        elif rating <= 2:
            _add_credit(reviewee, "bad_review", f"获得差评", order)
    
    return review
```

```python
        1. 获取已验证用户（JWTAuthMiddleware注入）
        2. 验证用户是否为会话参与者
        3. 加入 Channels group
        """
        self.user = self.scope.get("user")
        if not self.user or not self.user.is_authenticated:
            await self.close(code=4001)  # 未认证
            return
        
        self.conversation_uuid = self.scope["url_route"]["kwargs"]["conversation_uuid"]
        
        # 验证参与者身份
        is_participant = await self._is_participant(self.conversation_uuid, self.user)
        if not is_participant:
            await self.close(code=4003)  # 无权限
            return
        
        # 加入会话组
        self.group_name = f"chat_{self.conversation_uuid}"
        await self.channel_layer.group_add(self.group_name, self.channel_name)
        await self.accept()
    
    async def receive_json(self, content, **kwargs):
        """处理客户端消息类型：
        - chat_message: 发送聊天消息
        - mark_read: 标记消息已读
        - read_conversation: 标记整个会话已读
        - typing: 输入状态
        """
        msg_type = content.get("type")
        
        if msg_type == "chat_message":
            await self._handle_chat_message(content)
        elif msg_type == "mark_read":
            await self._handle_mark_read(content)
        elif msg_type == "read_conversation":
            await self._handle_read_conversation(content)
        elif msg_type == "typing":
            await self._handle_typing(content)
    
    async def _handle_chat_message(self, content):
        """处理聊天消息：保存到数据库并广播"""
        text = content.get("content", "").strip()
        if len(text) > 2000:
            await self.send_json({"type": "error", "error": "消息长度不能超过2000字"})
            return
        
        message = await self._save_message(text)
        
        # 广播给会话所有参与者
        await self.channel_layer.group_send(
            self.group_name,
            {
                "type": "chat_message",
                "message": {
                    "id": str(message["id"]),
                    "sender_id": str(self.user.id),
                    "sender_name": self.user.get_display_name(),
                    "content": text,
                    "created_at": message["created_at"].isoformat(),
                }
            }
        )
```

```python
@property
def is_trusted_seller(self) -> bool:
    # 基本条件
    if not self.is_active or self.credit_score < 150:
        return False
    
    # 180天内无违规记录
    recent_violation = CreditRecord.objects.filter(
        user=self,
        reason=CreditRecord.ChangeReason.VIOLATION,
        created_at__gte=timezone.now() - timedelta(days=180)
    ).exists()
    if recent_violation:
        return False
    
    # 完成交易≥3单
    completed = Order.objects.filter(
        seller=self, status=Order.Status.COMPLETED
    ).count()
    return completed >= 3
```
